-  rwx  r-x  r--   vichu  devops  file.txt
│   │    │    │
│   │    │    └─ others permissions
│   │    └─ group permissions
│   └─ owner (user) permissions
└─ file type: - =file  d =directory  l =symlink

r = read    = 4
w = write   = 2
x = execute = 1

rwx = 7   (read + write + execute)
rw- = 6   (read + write)
r-x = 5   (read + execute)
r-- = 4   (read only)
--- = 0   (no permissions)

chmod 777    rwxrwxrwx — everyone full access (avoid in production!)
chmod 755    rwxr-xr-x — owner full, others read+execute (web files)
chmod 644    rw-r--r-- — owner read+write, others read only (config files)
chmod 600    rw------- — owner only (SSH private keys)
chmod 700    rwx------ — owner only with execute (private scripts)
chmod 400    r-------- — read only, no one can modify (backups)

chmod +x file             Add execute for everyone
chmod u+x file            Add execute for owner only
chmod g-w file            Remove write from group
chmod o-r file            Remove read from others
chmod a+r file            Add read for all
chmod u=rwx,g=rx,o=r file Set exact permissions
chmod -R 755 /var/www     Recursively change all files in directory

ls -l filename            Show permissions of a file
ls -la                    Show all files including hidden with permissions
stat filename             Detailed info including permissions in octal

sudo chown vichu file.txt              Change owner to vichu
sudo chown vichu:devops file.txt       Change owner and group
sudo chown -R vichu:devops /var/www    Recursively change owner and group
sudo chgrp devops file.txt             Change only the group

umask                     Show current default permission mask
umask 022                 New files = 644, new dirs = 755 (default)
umask 027                 New files = 640, new dirs = 750 (stricter)

getfacl file.txt                    View ACL permissions
setfacl -m u:vichu:rw file.txt      Give vichu read+write access
setfacl -m g:devops:r file.txt      Give devops group read access
setfacl -x u:vichu file.txt         Remove ACL entry for vichu
setfacl -b file.txt                 Remove all ACL entries

lsattr file.txt                     View extended file attributes
sudo chattr +i file.txt             Make file immutable (even root can't delete)
sudo chattr -i file.txt             Remove immutable flag

sudo visudo                         Safely edit sudoers file
sudo -l                             List sudo commands current user can run

sudoers file format:
vichu ALL=(ALL:ALL) ALL
  │     │    │    └─ all commands
  │     │    └─ as any group
  │     └─ as any user
  └─ on all hosts

find / -perm 777           Find all fully open files (security risk)
find / -perm /4000         Find all SUID files (security audit)
find / -user vichu         Find all files owned by vichu
find / -nouser             Find orphaned files with no valid owner
